Example
new RGBA()
  .set( 'magenta' )                    // -> 255, 0, 255, 1
  .set( '#ff00ffff' )                  // -> 255, 0, 255, 1
  .set( '#ff00ff' )                    // -> 255, 0, 255, 1
  .set( '#f007' )                      // -> 255, 0, 0, 0.47
  .set( '#f00' )                       // -> 255, 0, 0, 1
  .set( 'hsla( 0, 100%, 50%, 0.47 )' ) // -> 255, 0, 0, 0.47
  .set( 'rgb( 0, 0, 0 )' )             // -> 0, 0, 0, 1
  .set( 0 )                            // -> 0, 0, 0, 1
  .set( 0, 0, 0 )                      // -> 0, 0, 0, 1
  .set( 0, 0 )                         // -> 0, 0, 0, 0
  .set( 0, 0, 0, 0 );                  // -> 0, 0, 0, 0
